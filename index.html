import time import random words = [ "python", "grok", "xai", "keyboard", "coffee", "sunshine", "galaxy", "rocket", "banana", "ninja", "pizza", "dragon", "zombie", "unicorn" ] def typing_race(): print("\n=== TYPING SPEED CHALLENGE ===") print("Type each word as fast as you can!\n") score = 0 total_time = 0 rounds = 10 for i in range(rounds): word = random.choice(words) print(f"[{i+1}/{rounds}] Type: {word}") start = time.time() typed = input("> ").strip() end = time.time() took = end - start if typed.lower() == word.lower(): print(f"✅ Correct! ({took:.2f}s)") score += 1 total_time += took else: print(f"❌ Wrong! It was '{word}'") avg_time = total_time / rounds if rounds > 0 else 0 print(f"\nGame Over! Score: {score}/{rounds} | Avg time per word: {avg_time:.2f}s") while True: typing_race() again = input("\nAnother round? (y/n): ").lower() if again != 'y': break
